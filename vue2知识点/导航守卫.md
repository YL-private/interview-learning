## 全局守卫 常用于权限判断
1. beforeEach 应用进入之前
<script>
  router.beforeEach(from, to, next) {
    if (to.query.token) {
      setToken(to.query.token);
    }
    if (getToken()) {
      /* has token*/
      if (to.path === "/ssoauthorize") {
        next();
      }
    }
  }
</script>

2. afterEach 路由进入之后
<script>
  router.afterEach(() => {
    done()
  });
</script>
